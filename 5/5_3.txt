SELECT
	*
FROM
	numbers AS N1
WHERE
	N1.number <> 1
AND
	NOT EXISTS
		(
			SELECT
				*
			FROM
				numbers AS N2
			WHERE
				N2.number <> 1
			AND
				N1.number > N2.number
			AND
				MOD(N1.number, N2.number) = 0	
		)
;
