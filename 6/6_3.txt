SELECT
	SI.shop,
	COUNT(I.item) AS my_item_cnt,
	(SELECT COUNT(item) FROM items) - COUNT(I.item) AS diff_cnt
FROM
	shop_items AS SI
LEFT OUTER JOIN
	items AS I
ON
	SI.item = I.item
GROUP BY
	SI.shop
;

	*
FROM

	items AS I
ON
	SI.item = I.item
