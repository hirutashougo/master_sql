MariaDBにおいては、MERGEが使用できないため、
INSERT INTOを用いて同様の処理を再現する。

/*class_Aテーブルに対し、
	class_Bテーブルから重複しない形で情報を挿入する*/
INSERT INTO
	class_A
	(
		id,
		name
	) 
SELECT
	id,
	name
FROM
	class_B
ON DUPLICATE KEY UPDATE 
	id = class_B.id,
	name = class_B.name
;

出力結果：
select * from class_A;
+----+--------+
| id | name   |
+----+--------+
|  1 | 田中   |
|  2 | 内海   |
|  3 | 伊集院 |
|  4 | 西園寺 |
+----+--------+
